def str_out
  if get_type == "IN"
    return "\n\nTransaction Type: #{get_type}\nAmount: +#{get_amount}$\nTime of Transaction: #{get_time}"
  else
    return "\n\nTransaction Type: #{get_type}\nAmount: -#{get_amount}$\nTime of Transaction: #{get_time}"
  end
end
